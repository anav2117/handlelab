// Añade esta lógica a tu script existente function decir(texto) { // Cancelar cualquier audio pendiente para que sea reactivo window.speechSynthesis.cancel(); const msg = new SpeechSynthesisUtterance(texto); msg.lang = (currentLang === 'en-US') ? 'en-US' : 'es-ES'; // Configuración de voz masculina grave msg.rate = 1.0; msg.pitch = 0.7; // Tono grave para una voz de coach autoritaria // Intentar buscar una voz masculina específica const voces = window.speechSynthesis.getVoices(); const vozPreferida = voces.find(v => v.name.includes('Male') || v.name.includes('David') || v.name.includes('Google US English') ); if (vozPreferida) msg.voice = vozPreferida; window.speechSynthesis.speak(msg); } // Actualizamos la función toggleRhythm para incluir el feedback inicial function toggleRhythm() { const btn = document.getElementById('ctrl-btn'); if (isPlaying) { clearInterval(interval); btn.innerText = (currentLang === 'en-US') ? "START ENGINE" : "INICIAR MOTOR"; } else { interval = setInterval(playSound, (60000 / bpm)); btn.innerText = (currentLang === 'en-US') ? "STOP ENGINE" : "PARAR MOTOR"; // Feedback de voz al iniciar const saludo = (currentLang === 'en-US') ? "Let's work. Focus on the rhythm." : "A trabajar. Enfócate en el ritmo."; decir(saludo); } isPlaying = !isPlaying; } // Y al cambiar la velocidad, el coach te avisa function setSpeed(val, btn) { bpm = val; document.querySelectorAll('.speed-btn').forEach(b => b.classList.remove('active')); btn.classList.add('active'); // Feedback de cambio de velocidad if (isPlaying) { const mensaje = (currentLang === 'en-US') ? "New tempo set." : "Nuevo ritmo establecido."; decir(mensaje); // Reiniciar el intervalo con la nueva velocidad clearInterval(interval); interval = setInterval(playSound, (60000 / bpm)); } }
